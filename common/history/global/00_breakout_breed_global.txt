# This is executed last among all history
GLOBAL = {
	every_state = {
		if = {
			limit = { state_population > 30000000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 2000
				days = 3
				is_decaying = yes
			}
		}
		if = {
			limit = { state_population > 10000000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 700
				days = 3
				is_decaying = yes
			}
		}
		else_if = {
			limit = { state_population > 3000000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 200
				days = 3
				is_decaying = yes
			}
		}
		else_if = {
			limit = { state_population > 1000000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 70
				days = 3
				is_decaying = yes
			}
		}
		else_if = {
			limit = { state_population > 300000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 20
				days = 3
				is_decaying = yes
			}
		}
		else_if = {
			limit = { state_population > 100000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 7
				days = 3
				is_decaying = yes
			}
		}
		else_if = {
			limit = { state_population > 30000 }
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 2
				days = 3
				is_decaying = yes
			}
		}
		else = {
			add_modifier = {
				name = breakout_breed_initial
				multiplier = 1
				days = 3
				is_decaying = yes
			}
		}
		if = {
			limit = {
				total_urbanization > 1000
			}
			create_building={
				building="breed_center"
				level = 15
				reserves = 0
			}
			if = {
				limit = {
					owner ?= {
						has_technology_researched = banking
					}
				}
				create_building={
					building="bank"
					level = 1
					reserves = 0
				}
			}
		}
		else_if = {
			limit = {
				total_urbanization > 500
			}
			create_building = {
				building="breed_center"
				level = 10
				reserves =0
			}
		}
		else_if = {
			limit = {
				total_urbanization > 200
			}
			create_building={
				building="breed_center"
				level = 5
				reserves = 0
			}
		}
		else_if = {
			limit = {
				total_urbanization >= 120
			}
			create_building={
				building="breed_center"
				level = 2
				reserves = 0
			}
		}
		else_if = {
			limit = {
				total_urbanization >= 95
			}
			create_building={
				building="breed_center"
				level = 1
				reserves = 0
			}
		}
	}
}
