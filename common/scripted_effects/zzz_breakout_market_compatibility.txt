add_good_merchant_profit = {
	if = {
		limit = {
			sg:$GOOD$.state_goods_pricier > market.mg:$GOOD$.market_goods_pricier
		}
		set_variable = {
			name = price_diff
			value = sg:$GOOD$.state_goods_pricier 
		}
		change_variable = {
			name = price_diff
			subtract = market.mg:$GOOD$.market_goods_pricier
		}
		change_variable = {
			name = price_diff
			multiply = sg:$GOOD$.state_goods_consumption
		}
	}
	else_if = {
		limit = {
			sg:$GOOD$.state_goods_pricier < market.mg:$GOOD$.market_goods_pricier
		}
		set_variable = {
			name = price_diff
			value = market.mg:$GOOD$.market_goods_pricier
		}
		change_variable = {
			name = price_diff
			subtract = sg:$GOOD$.state_goods_pricier 
		}
		change_variable = {
			name = price_diff
			multiply = sg:$GOOD$.state_goods_production
		}
	}
	change_variable = {
		name = total_merchant_profit
		add = var:price_diff
	}
}
